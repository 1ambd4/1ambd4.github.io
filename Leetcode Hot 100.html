    vector<int> twoSum(vector<int>& nums, int target) {
    int n = nums.size();

    vector<array<int, 2>> inums;
    for (int i = 0; i < n; ++i) inums.push_back({i, nums[i]});

    std::sort(inums.begin(), inums.end(), [](array<int, 2>& a, array<int, 2>& b) -> bool {
        return a[1] < b[1];
    });

    auto search = [&inums, n](int t) {
        int l = 0;
        int r = n - 1;

        while (l < r) {
            int m = (l + r + 1)>> 1;
            if (inums[m][1] <= t) l = m;
            else r = m - 1;
        }

        if (inums[l][1] == t) return l;
        else return -1;
    };

    for (int i = 0; i < n; ++i) {
        int t = target - inums[i][1];
        int j = search(t);
        if (j != -1 && j != i) return {inums[i][0], inums[j][0]};
    }

    return {};
}

    int longestConsecutive(vector<int>& nums) {

    int res = 0;

    for (const auto& x : nums) {
        if (m[x] == 0) {
            int l = m[x-1];
            int r = m[x+1];
            int d = l + 1 + r;

            m[x-l] = d;
            m[x+r] = d;
            m[x] = d;

            res = std::max(res, d);
        }
    }

    return res;
}

    vector<vector<int>> res;
    if (n == 0) return res;

    std::sort(nums.begin(), nums.end(), std::less<int>());

    auto search = [&nums, n](int t) -> int {
        int l = 0;
        int r = n - 1;
        while (l < r) {
            int m = (l + r + 1) >> 1;
            if (nums[m] <= t) l = m;
            else r = m - 1;
        }

        if (nums[l] == t) return l;
        else return -1;
    };

    for (int i = 0, j = 0; i < n; ++i) {
        for (j = i + 1; j < n; ++j) {
            int t = 0 - nums[i] - nums[j];
            int k = search(t);
            if (k != -1 && k > j) res.push_back({nums[i], nums[j], nums[k]});

        }
        while (i+1 < n && nums[i+1] == nums[i]) ++i;
    }

    return res;
}

    string minWindow(string s, string q) {
    int n = s.size();
    int m = q.size();

    int l = 0;
    int r = 0;
    int v = 0;
    int res = 0;

            if (wind[c] == need[c]) ++v;

        ++r;

        while (v == need.size()) {

                res = l;
                len = r - l;

            if (need.count(t)) {

                wind[t]--;
            }
            ++l;
        }
    }

    return len != n + 1 ? s.substr(res, len) : "";
}

    vector<vector<int>> merge(vector<vector<int>>& intervals) {
    int n = intervals.size();
    vector<vector<int>> res;

    std::sort(intervals.begin(), intervals.end(), [](vector<int>& a, vector<int>& b) -> bool {
        return a[0] < b[0];
    });

    for (int i = 0, j = 0; i < n; i = j) {
        int l = intervals[i][0];
        int r = intervals[i][1];

        if (i == n - 1) res.push_back({l, r});

        for (j = i + 1; j < n; ++j) {
            int u = intervals[j][0];
            int v = intervals[j][1];

            if (l <= u && u <= r) {

                r = std::max(r, v);
                if (j == n - 1) res.push_back({l, r});
            } else {
                res.push_back({l, r});
                break;
            }
        }
    }

    return res;
}
